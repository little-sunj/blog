# 2021.02.01 연습

&nbsp;

- 사용언어 : java

&nbsp;


```java

class Solution {
    public int countConsistentStrings(String allowed, String[] words) {
        
        HashSet<Character> set = new HashSet<>();
        int count = 0;
        boolean check = true;
        
        for (char c : allowed.toCharArray()) {
            set.add(c);
        }
        
        for (String s : words) {
            
            for (char sc : s.toCharArray()) {
                if (!set.contains(sc)) {
                    check = false;
                }
            }
            
            if (check) {count++;}
            check = true;
        }
        
        return count;
        
    }
}
```
